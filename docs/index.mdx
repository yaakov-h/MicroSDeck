# Quick Start

<CheckList>
	<CheckListItem check={(cardsAndGames) => cardsAndGames.length > 0}>
		Has Inserted a Card **bold** *italic* Stuff
	</CheckListItem>
	<CheckListItem check={(cardsAndGames) => cardsAndGames.filter(([_, games]) => games.length > 0).length > 0}>
		Has Installed a Game
	</CheckListItem>
	<CheckListItem check={(cardsAndGames) => cardsAndGames.filter(([card]) => card.name != "").length > 0}>
		Has Named a Card
	</CheckListItem>
	<CheckListItem check={(cardsAndGames) => cardsAndGames.filter(([_, games]) => games.filter(v => !v.is_steam) > 0).length > 0}>
		Has Added a Non Steam Game
	</CheckListItem>
</CheckList>
